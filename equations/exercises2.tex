\begin{problem}
Assuming none of the numbers involved are zero, select all operations below which must produce an irrational number.

\begin{selectAll}
	\choice{rational $+$ rational}
	\choice[correct]{rational $+$ irrational}
	\choice{irrational $+$ irrational}
	\choice{rational $\times$ rational}
	\choice[correct]{irrational $\times$ rational}
	\choice{irrational $\times$ irrational}
\end{selectAll}
\end{problem}

\begin{problem}
Write $\frac{1}{2+i}$ in the form $a + bi$.
\begin{prompt}
	$\answer[given]{\frac{2}{5}} + \answer[given]{-\frac{1}{5}}i$
\end{prompt}
\begin{hint}
Try multiplying the numerator and denominator by something that will make the denominator into a whole number (in other words, the complex conjugate of $2+i$).
\end{hint}

\end{problem}
